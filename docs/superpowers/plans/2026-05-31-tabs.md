# 멀티탭 (크롬 같은 탭) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development. Steps use checkbox (`- [ ]`) syntax.

**Goal:** 여러 문서를 탭으로 동시에 열어두고, 탭 바에서 전환·닫기하며, 각 탭은 자기 내용·커서·undo·dirty를 유지한다 (크롬 탭처럼).

**Architecture:** 순수 탭 메타 모델(`tabs.ts`: 탭 목록 + 활성 id, 각 탭 = path/savedText/currentText)을 두고, 탭별 **CodeMirror `EditorState`**를 `Map<id, EditorState>`에 보관해 `view.setState`로 스왑(탭별 undo/커서 보존). 탭 바 UI는 순수 DOM. 기존 단일 `docState`는 "활성 탭"으로 대체되고, 제목/상태바/이미지 docContext/자동저장은 활성 탭을 본다.

**Tech Stack:** CodeMirror 6 (`EditorState`/`view.setState`), Vitest. Rust 변경 없음.

> 범위: 탭 모델 · 탭 바 · 전환/열기/닫기 · 탭별 상태 보존 · 미저장 가드 · 활성 탭 자동저장. 탭 드래그 재정렬·탭 영속화(재시작 시 복원)는 추후.
>
> **시각 검증:** 탭 전환·닫기는 사람이 `npm run tauri dev`로 확인·반복.

```ts
import { describe, it, expect } from "vitest";
import { emptyTabs, openOrFocus, newUntitled, activeTab, updateActiveText, markActiveSaved, closeTab, tabDirty } from "./tabs";

describe("tabs model", () => {
  it("openOrFocus adds and activates a tab", () => {
    const s = openOrFocus(emptyTabs(), "/a.md", "A");
    expect(s.tabs.length).toBe(1);
    expect(activeTab(s)!.path).toBe("/a.md");
    expect(tabDirty(activeTab(s)!)).toBe(false);
  });
  it("opening the same path focuses the existing tab (no duplicate)", () => {
    let s = openOrFocus(emptyTabs(), "/a.md", "A");
    s = openOrFocus(s, "/b.md", "B");
    s = openOrFocus(s, "/a.md", "A");
    expect(s.tabs.length).toBe(2);
    expect(activeTab(s)!.path).toBe("/a.md");
  });
  it("updateActiveText makes the active tab dirty; markActiveSaved clears it", () => {
    let s = openOrFocus(emptyTabs(), "/a.md", "A");
    s = updateActiveText(s, "A edited");
    expect(tabDirty(activeTab(s)!)).toBe(true);
    s = markActiveSaved(s, "/a.md", "A edited");
    expect(tabDirty(activeTab(s)!)).toBe(false);
  });
  it("newUntitled adds an empty active tab with null path", () => {
    const s = newUntitled(emptyTabs());
    expect(activeTab(s)!.path).toBeNull();
  });
  it("closeTab removes and activates a neighbor; closing last leaves null", () => {
    let s = openOrFocus(emptyTabs(), "/a.md", "A");
    s = openOrFocus(s, "/b.md", "B");
    const aId = s.tabs[0].id;
    s = closeTab(s, s.activeId!);      // close active (b) -> neighbor a
    expect(s.tabs.length).toBe(1);
    expect(s.activeId).toBe(aId);
    s = closeTab(s, aId);
    expect(s.tabs.length).toBe(0);
    expect(s.activeId).toBeNull();
  });
});
```

```ts
export interface Tab { id: string; path: string | null; savedText: string; currentText: string; }
export interface TabsState { tabs: Tab[]; activeId: string | null; }

let counter = 0;
function newId(): string { counter += 1; return "t" + counter; }

export function emptyTabs(): TabsState { return { tabs: [], activeId: null }; }
export function activeTab(s: TabsState): Tab | null { return s.tabs.find((t) => t.id === s.activeId) ?? null; }
export function tabDirty(t: Tab): boolean { return t.currentText !== t.savedText; }

export function openOrFocus(s: TabsState, path: string, text: string): TabsState {
  const existing = s.tabs.find((t) => t.path === path);
  if (existing) return { ...s, activeId: existing.id };
  const tab: Tab = { id: newId(), path, savedText: text, currentText: text };
  return { tabs: [...s.tabs, tab], activeId: tab.id };
}

export function newUntitled(s: TabsState): TabsState {
  const tab: Tab = { id: newId(), path: null, savedText: "", currentText: "" };
  return { tabs: [...s.tabs, tab], activeId: tab.id };
}

export function setActive(s: TabsState, id: string): TabsState { return { ...s, activeId: id }; }

export function updateActiveText(s: TabsState, text: string): TabsState {
  return { ...s, tabs: s.tabs.map((t) => (t.id === s.activeId ? { ...t, currentText: text } : t)) };
}

/** 저장 스냅샷: savedText만 갱신(+path, save-as). currentText는 건드리지 않음(TOCTOU 안전). */
export function markActiveSaved(s: TabsState, path: string, savedText: string): TabsState {
  return { ...s, tabs: s.tabs.map((t) => (t.id === s.activeId ? { ...t, path, savedText } : t)) };
}

export function closeTab(s: TabsState, id: string): TabsState {
  const idx = s.tabs.findIndex((t) => t.id === id);
  if (idx < 0) return s;
  const tabs = s.tabs.filter((t) => t.id !== id);
  let activeId = s.activeId;
  if (s.activeId === id) activeId = (tabs[idx] ?? tabs[idx - 1])?.id ?? null;
  return { tabs, activeId };
}
```

```ts
import type { TabsState, Tab } from "./tabs";
import { tabDirty } from "./tabs";

export function mountTabBar(
  el: HTMLElement,
  handlers: { onSelect: (id: string) => void; onClose: (id: string) => void },
): { render: (s: TabsState) => void } {
  function title(t: Tab): string {
    if (!t.path) return "제목 없음";
    const i = Math.max(t.path.lastIndexOf("/"), t.path.lastIndexOf("\\"));
    return i >= 0 ? t.path.slice(i + 1) : t.path;
  }
  return {
    render(s) {
      el.replaceChildren();
      for (const t of s.tabs) {
        const tab = document.createElement("div");
        tab.className = "tab" + (t.id === s.activeId ? " active" : "");
        tab.addEventListener("click", (e) => { if (!(e.target as HTMLElement).closest(".close")) handlers.onSelect(t.id); });
        if (tabDirty(t)) {
          const dot = document.createElement("span"); dot.className = "dirty"; dot.textContent = "●"; tab.appendChild(dot);
        }
        const label = document.createElement("span"); label.className = "label"; label.textContent = title(t); tab.appendChild(label);
        const close = document.createElement("span"); close.className = "close"; close.textContent = "×";
        close.addEventListener("click", (e) => { e.stopPropagation(); handlers.onClose(t.id); });
        tab.appendChild(close);
        el.appendChild(tab);
      }
    },
  };
}
```
